//Comandos Git:

$ git --version    // Verificando se o Git esta instalado

$ git init   // Iniciando um repositório Git

$ git config --list // Verifica os valores padrões do usuário no github

$ git status   // Verificando o status dos arquivos

$ git add -A  // Adicionando arquivos

$ git commit -m "Primeiro Commit"   // Fazendo um Commit

$ git log  // Verificando se foi feito o commit e o hash

$ git remote add origin https://github.com/dev-alexandre-lopes/curso_python.git // Adicionando repositório remoto

$ git remote -v  // Verificando se o repositório remoto foi reconhecido

$ git push -u origin master  // Enviando o branch master local para o repositório remoto

$ git pull // Atualizar o repositório local

$ git rm -r --cached <nome arquivo> //Limpa o versionamento de um arquivo específico
